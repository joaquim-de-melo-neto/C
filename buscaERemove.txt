/* Esta função recebe uma lista encadeada lst com cabeça
* e remove da lista a primeira célula que contiver x ,
* se tal célula existir. */
void BuscaERemove (int x, célula *lst) {
célula *p, *q;
p = lst;
q = lst->seg;
while (q != NULL && q->conteúdo != x) {
p = q;
q = q->seg;
}
if (q != NULL) {
p->seg = q->seg;
free (q);
}
